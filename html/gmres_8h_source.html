36 template< class Matrix >
37 inline int
38 gmres( int m, int n, const Matrix &A, const double *b, double *x, double eps,
39  bool detailed ) {
40  if ( n<=0 )
41  return -1;
42  typedef double *doubleP;
43  double *V = new double[n*(m+1)];
44  double *U = new double[m*(m+1)/2];
45  double *r = new double[n];
46  double *y = new double[m+1];
47  double *c = new double[m];
48  double *s = new double[m];
49  double **v = new doubleP[m+1];
50  for ( int i=0; i<=m; ++i ) v[i]=V+i*n;
51  int its=-1;
52  {
53  double beta, h, rd, dd, nrm2b;
54  int j, io, uij, u0j;
55  nrm2b=dnrm2(n,b,1);
56 
57  io=0;
58  do { // "aussere Iteration
59  ++io;
60  mult(A,x,r);
61  daxpy(n,-1.,b,1,r,1);
62  beta=dnrm2(n,r,1);
63  dcopy(n,r,1,v[0],1);
64  dscal(n,1./beta,v[0],1);
65 
66  y[0]=beta;
67  j=0;
68  uij=0;
69  do { // innere Iteration j=0,...,m-1
70  u0j=uij;
71  mult(A,v[j],v[j+1]);
72  dgemv(Transpose,n,j+1,1.,V,n,v[j+1],1,0.,U+u0j,1);
73  dgemv(NoTranspose,n,j+1,-1.,V,n,U+u0j,1,1.,v[j+1],1);
74  h=dnrm2(n,v[j+1],1);
75  dscal(n,1./h,v[j+1],1);
76  for ( int i=0; i<j; ++i ) { // rotiere neue Spalte
77  double tmp = c[i]*U[uij]-s[i]*U[uij+1];
78  U[uij+1] = s[i]*U[uij]+c[i]*U[uij+1];
79  U[uij] = tmp;
80  ++uij;
81  }
82  { // berechne neue Rotation
83  rd = U[uij];
84  dd = sqrt(rd*rd+h*h);
85  c[j] = rd/dd;
86  s[j] = -h/dd;
87  U[uij] = dd;
88  ++uij;
89  }
90  { // rotiere rechte Seite y (vorher: y[j+1]=0)
91  y[j+1] = s[j]*y[j];
92  y[j] = c[j]*y[j];
93  }
94  ++j;
95  if ( detailed )
96  cout<<"gmres("<<m<<")\t"<<io<<"\t"<<j<<"\t"<<y[j]<<endl;
97  } while ( j<m && fabs(y[j])>=eps*nrm2b );
98  { // minimiere bzgl Y
99  dtpsv(UpperTriangle,NoTranspose,NotUnitTriangular,j,U,y,1);
100  // korrigiere X
101  dgemv(NoTranspose,n,j,-1.,V,n,y,1,1.,x,1);
102  }
103  } while ( fabs(y[j])>=eps*nrm2b );
104 
105  // R"uckgabe: Zahl der inneren Iterationen
106  its = m*(io-1)+j;
107  }
108 
109  delete[] V;
110  delete[] U;
111  delete[] r;
112  delete[] y;
113  delete[] c;
114  delete[] s;
115  delete[] v;
116  return its;
117 }
